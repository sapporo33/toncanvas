function draw() { /* RESET TRANSFORM */ ctx.setTransform(1, 0, 0, 1, 0, 0); /* CLEAR FULL CANVAS */ ctx.clearRect(0, 0, canvas.width, canvas.height); /* APPLY CAMERA */ ctx.setTransform(scale, 0, 0, scale, offsetX, offsetY); /* BACKGROUND */ ctx.fillStyle = "#1e293b"; ctx.fillRect(0, 0, GRID_SIZE * TILE_SIZE, GRID_SIZE * TILE_SIZE); /* GRID */ ctx.strokeStyle = "rgba(255,255,255,0.08)"; for (let x = 0; x < GRID_SIZE; x++) { for (let y = 0; y < GRID_SIZE; y++) { ctx.strokeRect(x*TILE_SIZE, y*TILE_SIZE, TILE_SIZE, TILE_SIZE); } } /* SELECTED */ ctx.fillStyle = "rgba(34,197,94,0.5)"; selectedTiles.forEach(key => { const [x, y] = key.split(',').map(Number); ctx.fillRect(x*TILE_SIZE, y*TILE_SIZE, TILE_SIZE, TILE_SIZE); }); /* HOVER */ if (hoverTile) { ctx.fillStyle = "rgba(59,130,246,0.3)"; ctx.fillRect(hoverTile.x*TILE_SIZE, hoverTile.y*TILE_SIZE, TILE_SIZE, TILE_SIZE); } /* SELECTION BOX */ if (isSelecting && selectStart && selectEnd) { const x1 = Math.min(selectStart.x, selectEnd.x); const y1 = Math.min(selectStart.y, selectEnd.y); const x2 = Math.max(selectStart.x, selectEnd.x); const y2 = Math.max(selectStart.y, selectEnd.y); ctx.fillStyle = "rgba(59,130,246,0.2)"; ctx.fillRect( x1*TILE_SIZE, y1*TILE_SIZE, (x2-x1+1)*TILE_SIZE, (y2-y1+1)*TILE_SIZE ); } }
